  <script>
    async function sendLogin() {
      const username = document.getElementById("username").value;
      const password = document.getElementById("password").value;
      const resultDiv = document.getElementById("result");

      resultDiv.innerHTML = '<span class="loading">Processing login request...</span>';

      try {
        const response = await fetch("https://9rz84c33j7.execute-api.eu-north-1.amazonaws.com/login", {
          method: "POST",
          headers: {
            "Content-Type": "application/json"
          },
          body: JSON.stringify({
            username: username,
            password: password
          })
        });

        const data = await response.json();

        let statusClass = "";
        if (data.login_status === "SUCCESS") {
          statusClass = "status-success";
        } else {
          statusClass = "status-failed";
        }

        let suspiciousText = data.suspicious
          ? '<span class="status-suspicious">Yes</span>'
          : 'No';

        resultDiv.innerHTML = `
          <p><strong>Message:</strong> ${data.message}</p>
          <p><strong>Login Status:</strong> <span class="${statusClass}">${data.login_status}</span></p>
          <p><strong>Suspicious:</strong> ${suspiciousText}</p>
          <p><strong>Reason:</strong> ${data.reason}</p>
        `;
      } catch (error) {
        resultDiv.innerHTML = `<span class="status-failed">Error: Could not connect to the API.</span>`;
      }
    }
  </script>
